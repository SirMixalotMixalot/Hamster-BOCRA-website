FILE 2: search.py
==================
This file takes a user's message and finds the best matching entries.
Create a new file called search.py and paste this EXACTLY:

---------- START OF FILE ----------
import json
import os

# Load the knowledge base once when the server starts
def load_knowledge_base():
    file_path = os.path.join(os.path.dirname(__file__), "knowledge_base.json")
    with open(file_path, "r", encoding="utf-8") as f:
        return json.load(f)

KNOWLEDGE_BASE = load_knowledge_base()


def search(message: str, top_n: int = 3):
    """
    Search the knowledge base for entries matching the user's message.
    Returns the top N matching entries sorted by relevance score.
    
    How scoring works:
    - Each keyword that appears in the message adds 1 point
    - If the message appears in the question text, add 3 bonus points
    - Higher score = better match
    """
    message_lower = message.lower()
    results = []

    for entry in KNOWLEDGE_BASE:
        score = 0

        # Check keywords
        for keyword in entry["keywords"]:
            if keyword.lower() in message_lower:
                score += 1

        # Bonus if message words appear in the question
        for word in message_lower.split():
            if len(word) > 2 and word in entry["question"].lower():
                score += 0.5

        if score > 0:
            results.append({"entry": entry, "score": score})

    # Sort by score (highest first) and return top N
    results.sort(key=lambda x: x["score"], reverse=True)
    return [r["entry"] for r in results[:top_n]]


def get_response(message: str) -> dict:
    """
    Main function: takes a user message, returns a chatbot response.
    """
    matches = search(message)

    if not matches:
        return {
            "reply": "I'm not sure about that, but I can help you get in touch with the right people. You can **call us** at +267 395 7755 or **email** info@bocra.org.bw. You can also submit a support ticket through your account.",
            "sources": []
        }

    if len(matches) == 1:
        entry = matches[0]
        return {
            "reply": entry["answer"],
            "sources": [{"title": entry["question"], "link": entry["link"]}]
        }

    # Multiple matches — combine the top answers
    reply_parts = []
    sources = []
    for i, entry in enumerate(matches):
        reply_parts.append(f"**{entry['question']}**\n{entry['answer']}")
        sources.append({"title": entry["question"], "link": entry["link"]})

    reply = "Here's what I found:\n\n" + "\n\n".join(reply_parts)
    return {"reply": reply, "sources": sources}

This is the server that receives chat messages and returns answers.

---------- START OF FILE ----------
from fastapi import FastAPI
from fastapi.middleware.cors import CORSMiddleware
from pydantic import BaseModel
from search import get_response

app = FastAPI(title="BOCRA Chatbot API")

app.add_middleware(
    CORSMiddleware,
    allow_origins=["http://localhost:8080", "http://localhost:5173"],
    allow_credentials=True,
    allow_methods=["*"],
    allow_headers=["*"],
)


class ChatRequest(BaseModel):
    message: str


class ChatSource(BaseModel):
    title: str
    link: str


class ChatResponse(BaseModel):
    reply: str
    sources: list[ChatSource]


@app.post("/api/ai/chat", response_model=ChatResponse)
async def chat(request: ChatRequest):
    result = get_response(request.message)
    return result


@app.get("/health")
async def health():
    return {"status": "ok"}
